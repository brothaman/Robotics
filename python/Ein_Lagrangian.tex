\begin{equation}
M_{i} = 
\begin{bmatrix}
m_{i} & 0 & 0 \\
0 & m_{i} & 0 \\
0 & 0 & m_{i} \\
\end{bmatrix}
\end{equation}

\begin{equation}
I_{b,i} = 
\begin{bmatrix}
I_{x} & 0 & 0 \\
0 & I_{y} & 0 \\
0 & 0 & I_{z} \\
\end{bmatrix}
\end{equation}



\begin{equation}
L & = & K & - & P
\end{equation}

\begin{equation}
K & = & \frac{M_{i}V_{i}^{2}}{2} = 
\frac{1}{2}\dot{q^{T}}\left[\sum\limits_{i=1}^{n} M_{i}J_{v,i}^{T}J_{v,i} + \\
J_{\omega,i}^{T}R_{i}I_{b,i}R_{i}^{T}J_{\omega,i}\right]\dot{q}
\end{equation} 

\begin{equation}
J_{i} = 
\begin{bmatrix}
J_{v,i} \\
J_{\omega,i}\\
\end{bmatrix}
=
\begin{bmatrix}
J_{v_{0}} & J_{v_{1}} & \dots & J_{v_{n}} \\
J_{\omega_{0}} & J_{\omega_{1}} & \dots & J_{\omega_{n}}
\end{bmatrix}
\end{equation}

\begin{equation}
P & = & M_{i}gh\\
= \sum\limits_{i=1}^{n} \vec{g}^{T} M_{i} \vec{r}_{c,i}
\end{equation}
